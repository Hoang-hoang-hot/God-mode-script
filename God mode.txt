local opened = false

--========================
-- DRAG OPEN BUTTON
--========================

local dragging = false
local dragInput
local dragStart
local startPos
local moved = false

local function updateDrag(input)

	local delta = input.Position - dragStart

	if math.abs(delta.X) > 5 or math.abs(delta.Y) > 5 then
		moved = true
	end

	openBtn.Position = UDim2.new(
		startPos.X.Scale,
		startPos.X.Offset + delta.X,
		startPos.Y.Scale,
		startPos.Y.Offset + delta.Y
	)

end

openBtn.InputBegan:Connect(function(input)

	if input.UserInputType == Enum.UserInputType.MouseButton1
	or input.UserInputType == Enum.UserInputType.Touch then

		dragging = true
		moved = false
		dragStart = input.Position
		startPos = openBtn.Position

		input.Changed:Connect(function()

			if input.UserInputState == Enum.UserInputState.End then
				dragging = false

				task.delay(0.1,function()
					moved = false
				end)

			end

		end)

	end

end)

openBtn.InputChanged:Connect(function(input)

	if input.UserInputType == Enum.UserInputType.MouseMovement
	or input.UserInputType == Enum.UserInputType.Touch then
		dragInput = input
	end

end)

UIS.InputChanged:Connect(function(input)

	if input == dragInput and dragging then
		updateDrag(input)
	end

end)

local function createSlider(text,default,min,max,y)

	local current = default

	local label = Instance.new("TextLabel")
	label.Parent = content
	label.Position = UDim2.new(0,15,0,y)
	label.Size = UDim2.new(0,240,0,20)
	label.BackgroundTransparency = 1
	label.Text = text..": "..default
	label.Font = Enum.Font.Gotham
	label.TextSize = 14
	label.TextColor3 = Color3.new(1,1,1)
	label.TextXAlignment = Enum.TextXAlignment.Left

	local minusBtn = Instance.new("TextButton")
	minusBtn.Parent = content
	minusBtn.Position = UDim2.new(0,15,0,y+22)
	minusBtn.Size = UDim2.new(0,22,0,22)
	minusBtn.Text = "-"
	minusBtn.Font = Enum.Font.GothamBold
	minusBtn.TextSize = 18
	minusBtn.TextColor3 = Color3.new(1,1,1)
	minusBtn.BackgroundColor3 = Color3.fromRGB(60,60,60)

	Instance.new("UICorner",minusBtn).CornerRadius = UDim.new(1,0)

	local plusBtn = Instance.new("TextButton")

	plusBtn.Parent = content
	plusBtn.Position = UDim2.new(0,233,0,y+22)
	plusBtn.Size = UDim2.new(0,22,0,22)
	plusBtn.Text = "+"
	plusBtn.Font = Enum.Font.GothamBold
	plusBtn.TextSize = 18
	plusBtn.TextColor3 = Color3.new(1,1,1)
	plusBtn.BackgroundColor3 = Color3.fromRGB(60,60,60)

	Instance.new("UICorner",plusBtn).CornerRadius = UDim.new(1,0)

	local bar = Instance.new("Frame")
	bar.Parent = content
	bar.Position = UDim2.new(0,45,0,y+28)
	bar.Size = UDim2.new(0,180,0,8)
	bar.BackgroundColor3 = Color3.fromRGB(70,70,70)

	Instance.new("UICorner",bar).CornerRadius = UDim.new(1,0)

	local fill = Instance.new("Frame")
	fill.Parent = bar
	fill.Size = UDim2.new(default/max,0,1,0)
	fill.BackgroundColor3 = Color3.fromRGB(255,255,255)
	fill.BorderSizePixel = 0

	Instance.new("UICorner",fill).CornerRadius = UDim.new(1,0)

	local knob = Instance.new("TextButton")
	knob.Parent = bar
	knob.Size = UDim2.new(0,18,0,18)
	knob.AnchorPoint = Vector2.new(0.5,0.5)
	knob.Position = UDim2.new(default/max,0,0.5,0)
	knob.Text = ""
	knob.BackgroundColor3 = Color3.new(1,1,1)

	Instance.new("UICorner",knob).CornerRadius = UDim.new(1,0)

	local sliderDragging = false

	local function refresh(percent)

		fill.Size = UDim2.new(percent,0,1,0)
		knob.Position = UDim2.new(percent,0,0.5,0)
		label.Text = text..": "..current

	end

	local function update(px)

		local percent = math.clamp(
			(px - bar.AbsolutePosition.X) / bar.AbsoluteSize.X,
			0,
			1
		)

		current = math.floor(min + ((max-min)*percent))
		refresh(percent)

	end

	minusBtn.MouseButton1Click:Connect(function()

		current = math.clamp(current - 1,min,max)
		refresh(current/max)

	end)

	plusBtn.MouseButton1Click:Connect(function()

		current = math.clamp(current + 1,min,max)
		refresh(current/max)

	end)

	knob.MouseButton1Down:Connect(function()
		sliderDragging = true
	end)

	UIS.InputEnded:Connect(function(input)

		if input.UserInputType == Enum.UserInputType.MouseButton1
		or input.UserInputType == Enum.UserInputType.Touch then
			sliderDragging = false
		end

	end)

	UIS.InputChanged:Connect(function(input)

		if sliderDragging and (
			input.UserInputType == Enum.UserInputType.MouseMovement
			or input.UserInputType == Enum.UserInputType.Touch
		) then
			update(input.Position.X)
		end

	end)

	bar.InputBegan:Connect(function(input)

		if input.UserInputType == Enum.UserInputType.MouseButton1
		or input.UserInputType == Enum.UserInputType.Touch then
			update(input.Position.X)
		end

	end)

	return function()
		return current
	end
end

local function addESP(plr)

	if plr == player then return end

	local function setup(char)

		if char:FindFirstChild("ESP_Highlight") then return end

		local head = char:WaitForChild("Head",5)
		if not head then return end

		local bill = Instance.new("BillboardGui")
		bill.Name = "ESP_Name"
		bill.Parent = head
		bill.Size = UDim2.new(0,160,0,40)
		bill.StudsOffset = Vector3.new(0,2.8,0)
		bill.AlwaysOnTop = true

		local txt = Instance.new("TextLabel")
		txt.Parent = bill
		txt.Size = UDim2.new(1,0,1,0)
		txt.BackgroundTransparency = 1
		txt.Text = plr.DisplayName
		txt.TextColor3 = Color3.new(1,1,1)
		txt.TextStrokeTransparency = 0
		txt.TextScaled = true
		txt.Font = Enum.Font.GothamBold

		local hl = Instance.new("Highlight")
		hl.Name = "ESP_Highlight"
		hl.Parent = char
		hl.FillTransparency = 1
		hl.OutlineTransparency = 0
		hl.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
		hl.OutlineColor = Color3.new(1,1,1)

	end

	if plr.Character then
		setup(plr.Character)
	end

	plr.CharacterAdded:Connect(setup)

end

local function removeESP(plr)

	if plr.Character then

		local char = plr.Character
		local head = char:FindFirstChild("Head")

		if head and head:FindFirstChild("ESP_Name") then
			head.ESP_Name:Destroy()
		end

		if char:FindFirstChild("ESP_Highlight") then
			char.ESP_Highlight:Destroy()
		end

	end

end

	if infiniteJump then

		local char = player.Character

		if char then

			local hum = char:FindFirstChildOfClass("Humanoid")

			if hum then
				hum:ChangeState(Enum.HumanoidStateType.Jumping)
			end

--========================
-- APPLY
--========================

local function apply()

	local char = player.Character
	if not char then return end

	local hum = char:FindFirstChildOfClass("Humanoid")
	if not hum then return end

	hum.WalkSpeed = getSpeed()
	hum.UseJumpPower = false
	hum.JumpHeight = getJump()

end

player.CharacterAdded:Connect(function()

	task.wait(1)
	apply()

end)

RunService.Stepped:Connect(function()

	apply()

	if noclip and player.Character then

		for _,v in pairs(player.Character:GetDescendants()) do

			if v:IsA("BasePart") then
				v.CanCollide = false
			end

		end

	end

end)
