local function apply()



local char = player.Character

if not char then return end

local hum = char:FindFirstChildOfClass("Humanoid")

if not hum then return end

local root = char:FindFirstChild("HumanoidRootPart")



if (sitEnabled or layEnabled) and root then

root.Anchored = true

hum.PlatformStand = true



local moveDirection = hum.MoveDirection

local camera = workspace.CurrentCamera

local isShiftLock = UIS.MouseBehavior == Enum.MouseBehavior.LockCenter



if isShiftLock and camera then

local camLook = camera.CFrame.LookVector * Vector3.new(1, 0, 1)

if camLook.Magnitude > 0.01 then targetLookVector = camLook.Unit end

elseif moveDirection.Magnitude > 0.01 then

targetLookVector = moveDirection.Unit

end



if sitEnabled then

root.CFrame = CFrame.lookAt(targetPosition, targetPosition + targetLookVector)

local torso = char:FindFirstChild("Torso")

if torso then

if torso:FindFirstChild("Left Hip") then torso["Left Hip"].Transform = CFrame.Angles(0, 0, math.rad(-90)) end

if torso:FindFirstChild("Right Hip") then torso["Right Hip"].Transform = CFrame.Angles(0, 0, math.rad(90)) end

if torso:FindFirstChild("Left Shoulder") then torso["Left Shoulder"].Transform = CFrame.new() end

if torso:FindFirstChild("Right Shoulder") then torso["Right Shoulder"].Transform = CFrame.new() end

end

if char:FindFirstChild("UpperTorso") then

local leftThigh = char:FindFirstChild("LeftUpperLeg") and char.LeftUpperLeg:FindFirstChild("LeftHip")

local rightThigh = char:FindFirstChild("RightUpperLeg") and char.RightUpperLeg:FindFirstChild("RightHip")

if leftThigh then leftThigh.Transform = CFrame.Angles(math.rad(-90), 0, 0) end

if rightThigh then rightThigh.Transform = CFrame.Angles(math.rad(-90), 0, 0) end

end

elseif layEnabled then

-- Xoay 90 độ để LƯNG ốp sát xuống mặt đất, kết hợp bộ đệm độ sâu của Torso

local baseCFrame = CFrame.lookAt(targetPosition, targetPosition + targetLookVector)

root.CFrame = baseCFrame * CFrame.Angles(math.rad(90), 0, 0)



local torso = char:FindFirstChild("Torso")

if torso then

if torso:FindFirstChild("Left Hip") then torso["Left Hip"].Transform = CFrame.new() end

if torso:FindFirstChild("Right Hip") then torso["Right Hip"].Transform = CFrame.new() end

if torso:FindFirstChild("Left Shoulder") then torso["Left Shoulder"].Transform = CFrame.new() end

if torso:FindFirstChild("Right Shoulder") then torso["Right Shoulder"].Transform = CFrame.new() end

end



if char:FindFirstChild("UpperTorso") then

for _, child in pairs(char:GetDescendants()) do

if child:IsA("Motor6D") and child.Name ~= "RootJoint" then

child.Transform = CFrame.new()

end

end

end

end

else

hum.WalkSpeed = getSpeed()

end



hum.UseJumpPower = false

hum.JumpHeight = getJump()

end



player.CharacterAdded:Connect(function() task.wait(1); apply() end)
